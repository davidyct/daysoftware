import matplotlib.pyplot as plt

def plot_results(vehicle, wheels, output, pavement, esubs, ths, layer_th, layer_str, results_str, results_wz):
    s = [('*' if ih == (layer_th - 1) else th) for ih, th in enumerate(pavement[0])]
    es = pavement[1][:-1] + ['*']
    ps = [float(str(p)[:6]) for p in pavement[2][:-1]] + ['*']
    title =  "Stress/Deflection" + '\n' + "Vehicle: " + str(vehicle) + '\n' + "Pavement: " + str(s) + ', ' + str(es) + ', ' + str(ps)

    plt.figure(figsize=(12.8, 9.6)) # width = 12.8 in, height = 9.6 in 

    plotter(esubs, "Subgrade, Modulus of Elasticity, psi", results_str, "Layer " + str(layer_str) + " stress, psi", ths, title, 211) # display title

    plotter(esubs, "Subgrade, Modulus of Elasticity, psi", results_wz, "Surface Deflection, in", ths, '', 212) # do not display title

    plt.show()

def plotter(x, xlabel, ys, ylabel, yls, title, id):
    pid = plt.subplot(id)
    for iy, yl in enumerate(yls): # plot each y for ys plots, yls labels
        pid.plot(x, ys[iy], label=str(yl))
    pid.set_title(title)
    pid.legend(loc="upper right")
    pid.set_xlabel(xlabel)
    pid.set_ylabel(ylabel)
